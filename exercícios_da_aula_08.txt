class Empregado:
  cont=0
  def __init__(self, id):
    self.id = id
    Empregado.cont=Empregado.cont+1
  def work(self):
    print(f'Funcionário id {self.id} está trabalhando')
   
emp1 = Empregado(100)
emp2 = Empregado(200)
emp1.work(), emp2.work()
print(f'O total de Funcionários trabalhando: {Empregado.cont}')

class Empregado:
    empCount = 0
    def __init__(self, nome, salario):
        self.nome = nome
        self.salario = salario
        Empregado.empCount = Empregado.empCount+1
    def nome(self):
        return self.nome 
    def salario(self):        
        return self.salario
emp1 = Empregado("Ana", 2000)
print(emp1.nome,emp1.salario) 
emp2 = Empregado("Maria", 5000)
print(emp2.nome,emp2.salario) 
print ("Total Empregados %d" % Empregado.empCount)
